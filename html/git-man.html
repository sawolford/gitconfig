git alias.man(1)

NAME

git man

SYNOPSIS

git man command - output the man page for a git command in the best format possible

DESCRIPTION

git man is a helper alias to output the man page for a git command in the best format available.

EXAMPLES

  • git man man
  • git man rebaseto

SEE ALSO

gitconfig

DETAILS

man = "!f() { : git man man; \
    [ $# -ne 1 ] && echo Usage: man command && return 1;                                    : Ensure proper number of arguments; \
    command() { sed '/D.*E.*T.*A.*I.*L.*S/q' | sed '$d' | grep -v '^$'; };                  : Command to prune off DETAILS section; \
    docpath=~/gitconfig/man;                                                                : Where man pages are kept; \
    htmlpath=~/gitconfig/html;                                                              : Where html pages are kept; \
    if git ingitbash; then                                                                  : Check if using git-bash; \
        docpath=$(cygpath -m $docpath);                                                     : Change to windows path with forward slashes; \
        htmlpath=$(cygpath -m $htmlpath);                                                   : Change to windows path with forward slashes; \
    fi;                                                                                     : Close if statement; \
    if git dohtml; then git web--browse \"$htmlpath/git-$1.html\";                          : If should do html, open web browser; \
    elif git haveman; then                                                                  : If should man pages, then; \
        export MANPATH=$docpath;                                                            : Set environment variable for man execution; \
        man git-$1 | tail -n +2 | command;                                                  : Execute man without uninteresting parts; \
    elif git havepandoc; then pandoc -f man \"$docpath/man1/git-$1.1\" -t plain | command;  : If pandoc is available, try to output man page; \
    else cat \"$docpath/man1/git-$1.1\" | sed 's,^\\.SH ,,' | grep -v '^\\.' | command; fi; : Cat the file as a last resort; \
} ; f"
